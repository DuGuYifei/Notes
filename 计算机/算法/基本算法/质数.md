        // 埃氏筛
        vector<int> isPrime(n, 1);
        int ans = 0;
        for (int i = 2; i < n; i++)
        {
            if (isPrime[i])
            {
                ans++;
                if((long)i * i < n)
                    for(int j = i * i; j<n; j += i)
                    {
                        isPrime[j]= 0;
                    }
                    // 因为 4*3  在 3*4 就算过了
            }
        }
        return ans;
    }
