# 发送_接收请求

## 流式请求

部署时肯能会有问题，需要:
[X-Accel-Buffering和Cache-Control](../../%E6%9C%8D%E5%8A%A1%E5%99%A8%E7%BD%91%E7%BB%9C/%E5%9F%9F%E5%90%8D%E5%92%8C%E7%BD%91%E5%9D%80%E4%BB%A5%E5%8F%8A%E8%B6%85%E6%96%87%E6%9C%AC%E4%BC%A0%E8%BE%93%E5%8D%8F%E8%AE%AE%E7%9A%84%E5%AE%9E%E9%99%85%E4%BD%BF%E7%94%A8/header.md)

```php
<?php
if ($_SERVER["REQUEST_METHOD"] == "POST") {
	$ch = curl_init();
	$inputData = file_get_contents('php://input');
	$jsonData = json_decode($inputData, true);
	if (json_last_error() !== JSON_ERROR_NONE) {
		echo json_encode(['error' => 'Invalid JSON data']);
		return;
	}
	$je = json_encode(
		array(
			"model" => "gpt-3.5-turbo",
			"max_tokens" => 400,
			"temperature" => .5,
			// "top_p" => 0.9, //"We generally recommend altering this or but not both.temperature"
			"frequency_penalty" => 0,
			"presence_penalty" => 0,
			"messages" => array(array(
				"role" => "user",
				"content" => $jsonData["prompt"]
			)),
			"stream" => true
		)
	);
	curl_setopt($ch, CURLOPT_URL, "https://api.openai.com/v1/chat/completions");
	curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
	curl_setopt($ch, CURLOPT_POST, 1);
	curl_setopt($ch, CURLOPT_POSTFIELDS, $je);
	curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
	curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, false);
	curl_setopt(
		$ch,
		CURLOPT_HTTPHEADER,
		array(
			"Content-Type: application/json",
			"Authorization: Bearer xxxxxxxxxxxxxxxxx"
		)
	);
	// -------------------------
	// handle the responce of stream
	curl_setopt($ch, CURLOPT_WRITEFUNCTION, function ($ch, $chunk) {
		$data_array = explode("data: ", $chunk);
		$i = 1;
		if(count($data_array) > 2)
			$i = 2;
		if($data_array[$i] == "[DONE]\n\n"){
			//ob_end_flush();
			return strlen($chunk);
		}
		$json_data = json_encode(substr($data_array[$i], 0, -2), true);
		str_replace("\"","\\\"", $json_data);
		$obj_data = json_decode(json_decode($json_data));
		$content = $obj_data->choices[0]->delta->content;
		if($content == null || $content == ""){
			return strlen($chunk);
		}
		echo $content;
		flush();
		ob_flush();
		return strlen($chunk);
	});
	// -------------------------
	//ob_start();
	//header('Content-Type: text/event-stream');
	//header('Cache-Control: no-cache');
	$response = curl_exec($ch);
	if ($response === false) {
		echo "cURL Error: " . curl_error($ch);
		curl_close($ch);
		return;
	}
}
?>
